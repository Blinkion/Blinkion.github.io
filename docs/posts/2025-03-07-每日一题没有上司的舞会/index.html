快要蓝桥杯比赛了，刚开学还不算特别忙，但是没有点鞭策也容易懈怠，通过写博客逼迫自己来看几眼算法知识（其实也是搞博客的直接动机），第一次先来一个简单的树形DP。

大意：树的每个节点有一个值，在不能同时取父和子节点的的前提下，求节点和的最大值。

#include <bits/stdc++.h>
using namespace std;
const int maxn = 6005;
vector<int> g[maxn];
int dp[maxn][2], a[maxn], n;//dp数组第二维0表示不选取，1表示选取
void dfs(int u, int fa)
{
    dp[u][0] = 0;
    dp[u][1] = a[u];
    for (auto x : g[u])
    {
        if (x != fa)
        {
            dfs(x, u);
            dp[u][0] += max(dp[x][1],dp[x][0]);
            dp[u][1] += dp[x][0];
        }
    }
}
int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    for (int i = 1; i <= n; i++)
    {
        int x, y;
        if(x==0 and y == 0)
            break;
        cin >> x >> y;
        g[x].emplace_back(y);
        g[y].emplace_back(x);
    }
    dfs(1, -1);//初始入口，根节点无父亲，任意赋值
    cout << max(dp[1][0], dp[1][1]);
}

链式前向星存图固然把空间利用到了极致，但在不卡stl时，使用vector+auto让代码更加简洁。笔者第一次做此题时，陷入了误区，没有考虑父子节点都不参加，结果谬以千里。
